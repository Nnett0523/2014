finding globers having a set of skills

  1. Build a map from skill id and level to (sorted) arrays of ids of globers having that skill at that level or higher
  2. Be requested the globers having a set of skills
  3. Get a set of n arrays, one for each skill in the set
  4. Find the intersection of the n arrays: that will be the globers fulfilling the request

array intersection

from: Simplest code for array intersection in javascript 

function intersect(array1, array2) {
   var result = [];
   // Don't destroy the original arrays
   var a = array1.slice(0);
   var b = array2.slice(0);
   var aLast = a.length - 1;
   var bLast = b.length - 1;
   while (aLast >= 0 && bLast >= 0) {
      if (a[aLast] > b[bLast] ) {
         a.pop();
         aLast--;
      } else if (a[aLast] < b[bLast] ){
         b.pop();
         bLast--;
      } else /* they're equal */ {
         result.push(a.pop());
         b.pop();
         aLast--;
         bLast--;
      }
   }
   return result;
}
