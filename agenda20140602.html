Control del cod de suscriptor:

   do
     echo "     Digite Codigo Suscriptor  ---> \c"
     read numsus
     cd $DATABASE
     nomsus=`grep "^$numsus" ICBSUS | cut -c9-48`
     echo "Codigo suscriptor: " $numsus  "   Suscriptor: " $nomsus
     if test -z "$nomsus"
     then
       echo "\tSuscriptor Errado"
     else
       loop=1
     fi
   done
